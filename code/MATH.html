MATH

4*MATH=HTAM

MATH.py

# N*[MATH]=[HTAM]
# MATH -> ABCD
# N(1000A+100B+10C+D)=1000D+100C+10B+A
# (1000N-1)A+10(10N-1)B=10(10-N)C+(1000-N)D
count = [0] * 9
for N in range(1, 10):
  for A in range(10):
    for B in range(10):
      for C in range(10):
        for D in range(10):
          if (1000*N-1)*A + 10*(10*N-1)*B == 10*(10-N)*C + (1000-N)*D:
            print("%d*%d%d%d%d=%d%d%d%d" % (N, A, B, C, D, D, C, B, A))
            count[N-1] += 1
print(count)
print("done")
